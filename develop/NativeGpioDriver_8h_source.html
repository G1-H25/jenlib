67 std::uint16_t analog_read(PinIndex pin) noexcept override {
68 auto vit = pin_voltage_volts_.find(pin);
69 if (vit != pin_voltage_volts_.end()) {
70 const std::uint16_t max_code =
71 static_cast<std::uint16_t>((1u << analog_read_bits_) - 1u);
72 float volts = vit->second;
73 if (volts < 0.0f) volts = 0.0f;
74 if (volts > reference_voltage_volts_) volts = reference_voltage_volts_;
75 const float ratio = reference_voltage_volts_ > 0.0f
76 ? (volts / reference_voltage_volts_)
77 : 0.0;
78 const unsigned int code = static_cast<unsigned int>(
79 ratio * static_cast<float>(max_code) + 0.5f);
80 return static_cast<std::uint16_t>(code);
81 }
82 auto it = analog_values_.find(pin);
83 return it != analog_values_.end() ? it->second : 0;
