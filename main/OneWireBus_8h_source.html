135// Inline CRC-8 (Dallas/Maxim, poly 0x31 reflected => 0x8C, init 0x00)
136template <typename InputIt>
137inline std::uint8_t OneWireBus::crc8(InputIt first, InputIt last) {
138 std::uint8_t crc = 0x00;
139 for (auto it = first; it != last; ++it) {
140 std::uint8_t in = static_cast<std::uint8_t>(*it);
141 for (int i = 0; i < 8; ++i) {
142 std::uint8_t mix = (crc ^ in) & 0x01;
143 crc >>= 1;
144 if (mix) crc ^= 0x8C; // reversed 0x31
145 in >>= 1;
146 }
147 }
148 return crc;
149}
